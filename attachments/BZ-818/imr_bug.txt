#include "hello_i.h"
const char * HelloWorld_oid = "HelloWorld";
int main(int argc, char * argv[])
{
    CORBA::ORB_var orb;
  try {
      // Initialize orb
      orb = CORBA::ORB_init(argc, argv);

      // Get reference to Root POA.
      CORBA::Object_var obj
          = orb->resolve_initial_references("RootPOA");
      PortableServer::POA_var poa
          = PortableServer::POA::_narrow(obj.in());
      // Get POA manager
      PortableServer::POAManager_var poa_mgr = poa->the_POAManager();
      /*
       *  Activate object
       */
      poa_mgr->activate();
      // Create a policy list. We use persistent objects with
      // user-assigned IDs, and explicit activation.
      CORBA::PolicyList policy_list;
      policy_list.length(2);
      policy_list[0] = poa->create_lifespan_policy(
                          PortableServer::PERSISTENT
                       );
      policy_list[1] = poa->create_id_assignment_policy(
                          PortableServer::USER_ID
                       );
      // Create a POA for HelloWorld
      PortableServer::POA_var HelloWorld_poa
          = poa->create_POA("HelloWorld", poa_mgr.in(), policy_list);
      // Create a sevant for the HelloWorld
      HelloWorld_impl HelloWorld_servant;
      // Create a reference for HelloWorld and
      // create the corresponding CORBA object.
      PortableServer::ObjectId_var oid
          = PortableServer::string_to_ObjectId(HelloWorld_oid);
      CORBA::Object_var ctrl
          = HelloWorld_poa->create_reference_with_id(
              oid.in(), "IDL:beaconnetworks.com/ISDX/HelloWorld:1.0"
            );
    HelloWorld_poa->activate_object_with_id(oid, &HelloWorld_servant);
      // Get reference to initial naming context.
      CosNaming::NamingContext_var inc
          = resolve_init<CosNaming::NamingContext>(
                  orb.in(), "NameService"
            );
      // Attempt to create CCS context.
      CosNaming::Name n;
      n.length(1);
      n[0].id = CORBA::string_dup("ISDX");
      try {
          CosNaming::NamingContext_var nc
              = inc->bind_new_context(n);
      } catch (const CosNaming::NamingContext::AlreadyBound &) {
          // Fine, HelloWorld context already exists.
      }
      // Force binding of controller reference to make
      // sure it is always up-to-date.
      n.length(2);
      n[1].id = CORBA::string_dup("HelloWorld");
      inc->rebind(n, ctrl.in());
      while (HelloWorld_servant.getState() != ISDX::STOPPING)
      {
        orb->perform_work();
      }
      inc->unbind(n);
      poa->destroy(1,1);  //  let the adaptor clean up, letting everything
finish
      orb->shutdown();    //  tell the ORB to shutdown
  }
  catch (const CORBA::Exception & e)
  {
      cerr << "Uncaught CORBA exception: " << e << endl;
      return 0;
  }
  catch (...)
  {
      assert(0);  // Uncaught exception, dump core
  }
  return 1;
}
